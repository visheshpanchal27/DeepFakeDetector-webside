            <div className="modern-card">
              <h2 className="text-xl font-bold text-gray-900 mb-4">Actions</h2>
              <div className="flex space-x-4">
                <button onClick={downloadReport} className="btn-primary flex-1">
                   Download Report
                </button>

                   Share Results
                </button>
              </div>
            </div>
          </div>
        </div>
      </motion.div>
    </div>
  );
